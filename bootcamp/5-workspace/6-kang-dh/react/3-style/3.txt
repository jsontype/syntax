sass : Syntactically Awesome Style Sheets
    // css
        body {
            font: 5px;
            color: red;
        }

    // sass
        $font-stack: Helvetica, sans-serif
        $primary-color: #333
        body
            font: 100% $font-stack
            color: $primary-color

    // scss : sass의 한 형태
        $font-stack: Helvetica, sans-serif;
        $primary-color: #333;
        body {
            font: 100% $font-stack;
            color: $primary-color;
        }
    
    // npm i node-sass
